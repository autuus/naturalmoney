CREATE TABLE  `naturalmoney`.`person` (
`id` INT NOT NULL AUTO_INCREMENT ,
`creation` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
`name` TEXT NOT NULL ,
`address` TEXT NOT NULL ,
`country` TEXT NOT NULL ,
`phone` TEXT NOT NULL ,
`birthdate` DATE NOT NULL ,
`social_security_number` TEXT NOT NULL ,
`email` TEXT NOT NULL ,
`active` BOOL DEFAULT NULL,
PRIMARY KEY (  `id` )
) ENGINE = MYISAM 


CREATE TABLE  `naturalmoney`.`account` (
`id` INT NOT NULL AUTO_INCREMENT ,
`balance` DOUBLE NOT NULL DEFAULT  '0',
`owner` INT NOT NULL ,
`login_ip` TEXT NOT NULL ,
`last_refresh` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
`username` TEXT NOT NULL ,
`password` TEXT NOT NULL ,
`active` BOOL DEFAULT NULL ,
UNIQUE (`id`)
) ENGINE = MYISAM 

CREATE TABLE  `naturalmoney`.`log` (
`id` INT NOT NULL AUTO_INCREMENT ,
`login` INT ,
`table` TEXT ,
`call` TEXT ,
`creation` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
PRIMARY KEY (  `id` )
) ENGINE = MYISAM ;

`to` INT ,
`from` INT ,

`creation` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
PRIMARY KEY (  `id` )
) ENGINE = MYISAM ;

CREATE TABLE  `naturalmoney`.`note` (
`id` INT NOT NULL AUTO_INCREMENT ,
`account` INT NOT NULL ,
`barcode` INT NOT NULL ,
`money` DOUBLE ,
`creation` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
PRIMARY KEY (  `id` )
) ENGINE = MYISAM ;
